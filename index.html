// pages/home/home.ts
quickCopy(e) {
  const recipeId = Number(e.currentTarget.dataset.id)
  const recipe = this.data.allRecipes.find(r => r.id === recipeId)
  const copied = { ...recipe, id: generateId(),
    name: recipe.name + '(复刻)', source: 'copy', draft: false }
  addRecipe(copied)                     // 本地Storage
  api.addRecipe({...copied, is_public:0}) // 云端私有
}

// 创建 — pages/restaurant/restaurant.ts
confirmCreateRest() {
  const code = 'SG' + Date.now().toString(36).toUpperCase().slice(-6)
  newRest = { inviteCode:code, codeExpire: Date.now()+86400000 }
  api.createRestaurant({...newRest, owner_id:0})
}
// 加入 — 本地优先，云端fallback
confirmJoinRest() {
  let target = rests.find(r => r.owner && r.inviteCode === input)
  if(!target) api.getRestaurantByCode(input)  // 云端
  doJoinRest(rests, target)  // 建originalId副本
}
// 点单 — restaurantId=originalId
submitOrder() {
  const order = { restaurantId: activeRest.originalId, status:'pending' }
  orders.unshift(order); saveOrders(orders)
}
// 订单显示 — 关联ID Set匹配
refreshAll() {
  const relatedIds = new Set([restId, activeRest.id])
  allRests.forEach(r => { if(r.originalId===activeRest.id) relatedIds.add(r.id) })
  filteredOrders = orders.filter(o => relatedIds.has(o.restaurantId))
}

// 双重匹配标记所有食客副本
deleteRestaurant() {
  rest.closed = true
  rests.forEach(r => {
    if(r.originalId===rest.id || r.inviteCode===rest.inviteCode) r.closed = true
  })
}
